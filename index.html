SQL inicial — Supabase → SQL Editor → Run
-- 1. Perfiles (agrega supervisor_area) alter table profiles drop constraint if exists profiles_rol_check; alter table profiles add constraint profiles_rol_check check (rol in ('tecnico','supervisor','supervisor_area','admin','gerencia')); -- Agrega columna zona al perfil (para supervisor_area) alter table profiles add column if not exists zona text; -- 2. Resto de tablas (si no existen) create table if not exists equipos ( id serial primary key, codigo text unique not null, nombre text not null, area text, modelo text, serie text, criticidad text default 'media', estado text default 'Operativo', created_at timestamptz default now() ); create table if not exists ordenes_trabajo ( id serial primary key, equipo_id int references equipos(id), equipo_nombre text, equipo_zona text, tipo text, prioridad text, estado text default 'Pendiente', tecnico_id uuid references profiles(id), tecnico_nombre text, descripcion text, observaciones text, tiempo_minutos int, fecha_creacion timestamptz default now(), fecha_cierre timestamptz, creado_por uuid references profiles(id) ); create table if not exists respuestas_checklist ( id serial primary key, orden_id int references ordenes_trabajo(id) on delete cascade, item_id text, item_label text, respuesta text, created_at timestamptz default now() ); -- 3. Realtime alter publication supabase_realtime add table ordenes_trabajo; -- 4. Equipos de ejemplo con zona insert into equipos (codigo,nombre,area,modelo,estado) values ('CNC-01','Torno CNC','Zona 1','Mazak QTN-200','Operativo'), ('COMP-02','Compresor Atlas','Zona 2','Atlas GA15','Operativo'), ('BAND-01','Banda transportadora','Zona 3','Intralox S1200','Operativo'), ('SOLD-03','Robot soldador','Zona 4','FANUC ARC-M','Operativo'), ('BOMB-01','Bomba hidráulica','Zona 5','Parker PV180','Operativo'), ('ELECT-01','Tablero eléctrico E3','Zona 6','Siemens S7-400','Operativo') on conflict (codigo) do nothing;
